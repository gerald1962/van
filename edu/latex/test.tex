\documentclass{article}
\usepackage[dvipsnames]{xcolor}  % Using the color package
\usepackage{stackengine}         % Stacking of objects:

\newlength\lunderset

% munderline - 

\newcommand\munderline[4][1]{%
  \setbox0 = \hbox{#2}%
  \stackunder[#1 \lunderset - #4]
             {\strut #2}
             {\color{#3} \rule{\wd0} {#4}}%
}

\newcommand\world[1] {#1}
\newcommand\wcase[1] {#1}
\newcommand\wspace[1] {#1}

\begin{document}
\lunderset = 1.25pt\relax

\munderline[1]{\world World: education} {red}{1pt}

\vskip 20pt
\munderline[2]{\munderline[1]{\wcase Case: x} {red}{1pt}}{green}{2pt}

\vskip 20pt
\munderline[3]{\munderline[2]{\munderline[1]{\wspace Space: y}{red}{1pt}}{green}{2pt}}{blue}{4pt}
\end{document}
