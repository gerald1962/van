% munderline - draw multiple underlines.
% #1: variable argument.
% #2: text.
% #3: underline color.
% #4: line width
\newcommand\munderline[4][1]{%
  \setbox0 = \hbox{#2}%
  \stackunder[#1 \lunderset - #4]
             {\strut #2}
             {\color{#3} \rule{\wd0} {#4}}%
}
